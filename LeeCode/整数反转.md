        var convertX: Int = x > 0 ? x : -x
        var rev:Int = 0;
        while (convertX != 0) {
            let pop:Int = convertX % 10;

class Solution {
    func reverse(_ x: inout Int) -> Int {
        var rev:Int = 0;
        while (x != 0) {
            let pop:Int = x % 10;
            x /= 10;
            if rev > INT_MAX/10 || (rev == INT_MAX / 10 && pop > 7) {
                return 0
            }
            if (rev < -INT_MAX/10 || (rev == -INT_MAX / 10 && pop < -8)) {
                return 0
            }
            rev = rev * 10 + pop;
        }
        return rev
    }
}
